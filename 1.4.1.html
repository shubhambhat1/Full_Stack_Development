const readline = require('readline'); const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); let employee =[]; function menu(){ console.log("1. Add Employee\n 2. Display Employee\n 3. Exit"); rl.question("Enter your choice: ", function(choice){ switch(choice){ case '1': addEmployee(); break; case '2': displayEmployee(); break; case '3': rl.close(); break; default: console.log("Invalid choice"); menu(); } }); } function addEmployee(){ rl.question("Enter employee name: ", function(name){ rl.question("Enter employee id: ", function(id){ employee.push({name: name, id: id}); console.log("Employee added successfully"); menu(); }); }); } function displayEmployee(){ if(employee.length === 0){ console.log("No employee found"); } else { console.log("Employee List:"); employee.forEach(emp => { console.log(`Name: ${emp.name}, ID: ${emp.id}`); } ); } menu(); } menu();
